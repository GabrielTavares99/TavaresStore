CREATE DATABASE bd_loja;

CREATE TABLE tb_produtos(
	id INTEGER AUTO_INCREMENT PRIMARY KEY
	,nome VARCHAR (255)
	,preco DECIMAL (10,2)
	,descricao TEXT
	,categoria_id INTEGER
	,usado BOOLEAN DEFAULT FALSE
);

ALTER TABLE tb_produtos ADD CONSTRAINT categoria_id FOREIGN KEY(id) REFERENCES tb_categorias(id)

CREATE TABLE tb_categorias(
	id INTEGER AUTO_INCREMENT PRIMARY KEY
	,nome NVARCHAR (255)
);

CREATE TABLE tb_usuarios(
	id INTEGER AUTO_INCREMENT PRIMARY KEY
	,email VARCHAR (255)
	,senha VARCHAR (255)
);
